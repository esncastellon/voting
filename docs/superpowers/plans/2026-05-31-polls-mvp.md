> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convertir la app actual en un MVP real de votaciones basado en `polls`, con CRUD, voto autenticado, usuarios/roles, dashboard y limpieza de restos del tutorial.

**Architecture:** Crear un modulo nuevo `app/lib/poll` y una UI nueva `app/ui/polls` para evitar arrastrar nombres `survey`. **Tech Stack:** Next.js App Router, React 19, NextAuth v5, postgres.js, zod, bcryptjs, Tailwind, MUI Date Pickers, Vitest para pruebas unitarias de reglas puras.

Create `app/lib/poll/validation.ts`:

```ts
import { z } from "zod";

const StatusSchema = z.enum(["draft", "open", "closed"]);

const OptionSchema = z.object({
  id: z.string().uuid().optional(),
  label: z.string().trim().min(1, "La opcion no puede estar vacia."),
  position: z.coerce.number().int().min(0),
});

export const PollInputSchema = z
  .object({
    id: z.string().uuid().optional(),
    title: z.string().trim().min(1, "El titulo es obligatorio."),
    description: z.string().trim().optional().default(""),
    question: z.string().trim().min(1, "La pregunta es obligatoria."),
    status: StatusSchema,
    starts_at: z.string().nullable().optional().default(null),
    ends_at: z.string().nullable().optional().default(null),
    options: z
      .array(OptionSchema)
      .transform((options) =>
        options
          .filter((option) => option.label.trim().length > 0)
          .sort((a, b) => a.position - b.position)
          .map((option, index) => ({ ...option, position: index })),
      )
      .refine((options) => options.length >= 2, {
        message: "La votacion necesita al menos dos opciones.",
      }),
  })
  .refine(
    (poll) =>
      !poll.starts_at ||
      !poll.ends_at ||
      new Date(poll.starts_at).getTime() <= new Date(poll.ends_at).getTime(),
    {
      message: "La fecha de inicio debe ser anterior a la fecha de cierre.",
      path: ["ends_at"],
    },
  );

export type PollInputData = z.infer<typeof PollInputSchema>;

export function validatePollInput(input: unknown) {
  return PollInputSchema.safeParse(input);
}
```

Create `app/lib/poll/vote-rules.ts`:

```ts
import type { PollStatus, RoleName } from "./definitions";

type PollVoteState = {
  status: PollStatus;
  starts_at: string | null;
  ends_at: string | null;
};

type VoteEligibilityInput = {
  poll: PollVoteState;
  roleName: RoleName | null | undefined;
  alreadyVoted: boolean;
  optionBelongsToPoll: boolean;
  now?: Date;
};

export function getVoteEligibility({
  poll,
  roleName,
  alreadyVoted,
  optionBelongsToPoll,
  now = new Date(),
}: VoteEligibilityInput) {
  if (roleName !== "usuario") {
    return { canVote: false, reason: "Tu rol no permite votar." };
  }

  if (poll.status !== "open") {
    return { canVote: false, reason: "La votacion no esta abierta." };
  }

  if (poll.starts_at && new Date(poll.starts_at) > now) {
    return { canVote: false, reason: "La votacion todavia no ha empezado." };
  }

  if (poll.ends_at && new Date(poll.ends_at) < now) {
    return { canVote: false, reason: "La votacion esta cerrada." };
  }

  if (alreadyVoted) {
    return { canVote: false, reason: "Ya has votado en esta votacion." };
  }

  if (!optionBelongsToPoll) {
    return { canVote: false, reason: "La opcion no pertenece a esta votacion." };
  }

  return { canVote: true, reason: null };
}
```

```ts
export function calculateParticipationRate({
  totalVotes,
  eligibleUsers,
  pollCount,
}: {
  totalVotes: number;
  eligibleUsers: number;
  pollCount: number;
}) {
  const denominator = eligibleUsers * pollCount;
  if (denominator === 0) return 0;
  return Math.round((totalVotes / denominator) * 100);
}
```
